# 创建子进程（雇佣工人）
process = subprocess.Popen(
    ["python", "-u", "worker.py"],  # 工人要执行的程序
    stdin=subprocess.PIPE,           # 你可以对他说话
    stdout=subprocess.PIPE,          # 你能听到他的回答
    stderr=subprocess.PIPE,          # 你能听到他的抱怨
    text=True                        # 使用文字交流（而不是二进制）
)

# 通过管道通信
process.stdin.write("开始工作\n")     # 通过管道说话
output = process.stdout.readline()   # 通过管道听回答

    stderr=subprocess.PIPE   # 将错误输出连接到管道
)

process = subprocess.Popen(
    ["grep", "error"],
    stdin=subprocess.PIPE,    # 通过管道提供输入
    stdout=subprocess.PIPE
)

# 3. 同时使用
process = subprocess.Popen(

    stdin=subprocess.PIPE,    # 输入管道
