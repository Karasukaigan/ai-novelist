from langgraph.graph import StateGraph, START
from langgraph.checkpoint.memory import InMemorySaver
from typing_extensions import TypedDict, NotRequired
import uuid

class State(TypedDict):
    topic: NotRequired[str]
    joke: NotRequired[str]


def generate_topic(state: State):
    return {"topic": "socks in the dryer"}


def write_joke(state: State):
    return {"joke": f"Why do {state['topic']} disappear? They elope!"}


checkpointer = InMemorySaver()
graph = (
    StateGraph(State)
    .add_node("generate_topic", generate_topic)
    .add_node("write_joke", write_joke)
    .add_edge(START, "generate_topic")
    .add_edge("generate_topic", "write_joke")
    .compile(checkpointer=checkpointer)
)

# Step 1: Run the graph
config = {"configurable": {"thread_id": str(uuid.uuid4())}}
result = graph.invoke({}, config)

# Step 2: Find a checkpoint to replay from
history = list(graph.get_state_history(config))
# History is in reverse chronological order
for state in history:
    print(f"next={state.next}, checkpoint_id={state.config['configurable']['checkpoint_id']}")

# Step 3: Replay from a specific checkpoint
# Find the checkpoint before write_joke
before_joke = next(s for s in history if s.next == ("write_joke",))

# write_joke re-executes (runs again), generate_topic does not

根据我查看的 LangGraph 源码，我来详细解释这些机制：

1. MessagesState 的自动添加机制
MessagesState 的定义是：

class MessagesState(TypedDict):
    messages: Annotated[list[AnyMessage], add_messages]

关键在于 Annotated[list[AnyMessage], add_messages]，其中 add_messages 是一个 reducer 函数（约简函数）。

当节点返回 {"messages": [response]} 时（如 graph_builder.py:179），LangGraph 会自动调用 add_messages 函数来合并新消息到现有消息列表。

2. add_messages 函数的行为
add_messages 的核心逻辑：

def add_messages(left: Messages, right: Messages, ...) -> Messages:
    # 合并两个消息列表
    merged = left.copy()
    merged_by_id = {m.id: i for i, m in enumerate(merged)}
    
    for m in right:
        if (existing_idx := merged_by_id.get(m.id)) is not None:
            # 如果 ID 相同，替换旧消息
            merged[existing_idx] = m
        else:
            # 如果 ID 不同，追加新消息
            merged_by_id[m.id] = len(merged)
            merged.append(m)

关键规则：

默认是追加模式：新消息会追加到现有消息列表后面
ID 相同则替换：如果新消息的 ID 与现有消息的 ID 相同，则替换该消息
自动分配 ID：如果消息没有 ID，会自动分配一个 UUID
3. graph.invoke 的行为
graph.invoke(input) 的处理流程：

初始状态：如果 input 包含 messages 键，这些消息会被设置为初始状态
节点返回：每个节点返回的 {"messages": [new_message]} 会通过 add_messages 自动追加
累积：随着节点执行，消息会不断累积
示例：

# 初始输入
graph.invoke({"messages": [HumanMessage("你好")]})
# 状态: [HumanMessage("你好")]

# 节点返回 {"messages": [AIMessage("你好！")]}
# 状态: [HumanMessage("你好"), AIMessage("你好！")]

4. graph.update_state 的行为
从 pregel/main.py:2309-2320 可以看到，update_state 同样使用 add_messages 来合并消息：

def update_state(self, config, values, as_node=None, task_id=None):
    """Update the state of the graph with the given values"""
    return self.bulk_update_state(config, [[StateUpdate(values, as_node, task_id)]])

关键点：

追加而非覆盖：graph.update_state 对 messages 字段是追加操作，不是覆盖
同样使用 add_messages：新消息会被追加到现有消息列表中
ID 相同则替换：如果消息 ID 相同，会替换对应的消息
总结对比
操作	messages 字段行为
graph.invoke({"messages": [...]})	设置初始消息列表
节点返回 {"messages": [...]}	追加到现有消息列表
graph.update_state(config, {"messages": [...]})	追加到现有消息列表（不是覆盖！）
消息 ID 相同时	替换对应的消息
如果需要覆盖所有消息，需要使用 RemoveMessage 特殊消息类型：

from langchain_core.messages import RemoveMessage

# 删除所有消息
graph.update_state(config, {"messages": [RemoveMessage(id="__remove_all__")]})

# 然后添加新消息
graph.update_state(config, {"messages": [new_message]})
